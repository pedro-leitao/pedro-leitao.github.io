# Define the productivity model
def calculate_unproductive_hours(
    team_size,
    weeks,
    context_switching_factor=0.05,
    communication_factor=0.03,
    meeting_factor=0.1,
    integration_factor=0.02,
    onboarding_factor=0.15,
):
    """
    Calculate unproductive hours based on team size and various overhead factors.

    Parameters:
    -----------
    team_size : int
        Number of people in the team
    weeks : int
        Project duration in weeks
    context_switching_factor : float
        Overhead due to context switching (per person per week)
    communication_factor : float
        Overhead due to communication complexity (scales with N^2)
    meeting_factor : float
        Overhead due to meetings (per person per week)
    integration_factor : float
        Overhead due to integration work (scales with N)
    onboarding_factor : float
        Overhead due to onboarding (per new team member)

    Returns:
    --------
    dict
        Dictionary with unproductive hours breakdown
    """
    # Assuming 40 productive hours per week per person
    total_hours = team_size * weeks * 40

    # Context switching overhead (increases with team size)
    context_switching_hours = team_size * weeks * 40 * context_switching_factor

    # Communication overhead (scales approximately with n^2 - more people means more communication channels)
    communication_channels = (team_size * (team_size - 1)) / 2
    communication_hours = communication_channels * weeks * communication_factor * 40

    # Meeting overhead (scales linearly with team size, but can be highly variable)
    meeting_hours = team_size * weeks * 40 * meeting_factor

    # Integration overhead (more people means more code to integrate)
    integration_hours = team_size * weeks * 40 * integration_factor

    # Onboarding overhead (new team members need time to onboard)
    # Simplified model: assume onboarding impact lasts for 4 weeks or the project duration, whichever is shorter
    onboarding_weeks = min(4, weeks)
    onboarding_hours = team_size * onboarding_weeks * 40 * onboarding_factor

    # Total unproductive hours
    total_unproductive = (
        context_switching_hours
        + communication_hours
        + meeting_hours
        + integration_hours
        + onboarding_hours
    )

    # Calculate productive hours
    productive_hours = total_hours - total_unproductive

    # Calculate productivity percentage
    productivity_percentage = (productive_hours / total_hours) * 100

    return {
        "team_size": team_size,
        "weeks": weeks,
        "total_hours": total_hours,
        "context_switching_hours": context_switching_hours,
        "communication_hours": communication_hours,
        "meeting_hours": meeting_hours,
        "integration_hours": integration_hours,
        "onboarding_hours": onboarding_hours,
        "total_unproductive": total_unproductive,
        "productive_hours": productive_hours,
        "productivity_percentage": productivity_percentage,
    }


# Function to create 3D visualization of the productivity model
def create_3d_viz(
    max_team_size=30,
    max_weeks=26,
    context_switching_factor=0.05,
    communication_factor=0.03,
    meeting_factor=0.1,
    integration_factor=0.02,
    onboarding_factor=0.15,
):
    """Create 3D visualization of productivity model using pythreejs"""

    # Generate data for the 3D surface
    team_sizes = np.arange(1, max_team_size + 1)
    week_values = np.arange(1, max_weeks + 1)

    team_grid, week_grid = np.meshgrid(team_sizes, week_values)

    productivity_values = np.zeros(team_grid.shape)

    # Calculate productivity percentage for each combination
    for i in range(len(week_values)):
        for j in range(len(team_sizes)):
            result = calculate_unproductive_hours(
                team_sizes[j],
                week_values[i],
                context_switching_factor,
                communication_factor,
                meeting_factor,
                integration_factor,
                onboarding_factor,
            )
            productivity_values[i, j] = result["productivity_percentage"]

    # Create 3D visualization
    # Convert the data to a format suitable for pythreejs
    x_positions = team_grid.flatten()
    y_positions = week_grid.flatten()
    z_positions = productivity_values.flatten()

    # Normalize values for color mapping
    colors = np.zeros((len(z_positions), 3))
    norm = plt.Normalize(z_positions.min(), z_positions.max())

    # Use a colormap (blue to red)
    cmap = plt.cm.coolwarm

    for i in range(len(z_positions)):
        colors[i] = cmap(norm(z_positions[i]))[:3]

    # Create geometries for each data point
    geometries = []
    materials = []
    meshes = []

    for i in range(len(x_positions)):
        geometry = BoxBufferGeometry(0.8, 0.8, z_positions[i] / 10)
        material = MeshLambertMaterial(color=tuple(colors[i]))
        mesh = Mesh(geometry=geometry, material=material)
        mesh.position = [x_positions[i], y_positions[i], z_positions[i] / 20]
        geometries.append(geometry)
        materials.append(material)
        meshes.append(mesh)

    # Create scene
    scene = Scene()

    # Add all meshes to the scene
    for mesh in meshes:
        scene.add(mesh)

    # Add lighting
    scene.add(AmbientLight(color="#ffffff", intensity=0.5))
    directional_light = DirectionalLight(
        color="#ffffff", position=[10, 10, 10], intensity=0.5
    )
    scene.add(directional_light)

    # Set up camera
    camera = PerspectiveCamera(
        position=[max_team_size / 2, max_weeks / 2, 50],
        fov=75,
        aspect=1.0,
        near=0.1,
        far=1000,
    )

    # Set up controls
    controls = OrbitControls(controlling=camera)

    # Create renderer
    renderer = Renderer(
        camera=camera, scene=scene, controls=[controls], width=800, height=600
    )

    return renderer


# Function to create interactive 2D visualization
def plot_interactive_productivity():
    """Create interactive 2D plots for productivity model"""

    # Define the layout for sliders to make them wider
    slider_layout = Layout(width="600px")

    @interact(
        team_size=IntSlider(
            min=1,
            max=50,
            step=1,
            value=10,
            description="Team Size:",
            layout=slider_layout,
        ),
        weeks=IntSlider(
            min=1,
            max=52,
            step=1,
            value=12,
            description="Project Weeks:",
            layout=slider_layout,
        ),
        context_switching_factor=FloatSlider(
            min=0,
            max=0.2,
            step=0.01,
            value=0.05,
            description="Context Switching Overhead:",
            layout=slider_layout,
        ),
        communication_factor=FloatSlider(
            min=0,
            max=0.1,
            step=0.005,
            value=0.03,
            description="Communication Overhead:",
            layout=slider_layout,
        ),
        meeting_factor=FloatSlider(
            min=0,
            max=0.3,
            step=0.01,
            value=0.1,
            description="Meeting Overhead:",
            layout=slider_layout,
        ),
        integration_factor=FloatSlider(
            min=0,
            max=0.1,
            step=0.005,
            value=0.02,
            description="Integration Overhead:",
            layout=slider_layout,
        ),
        onboarding_factor=FloatSlider(
            min=0,
            max=0.5,
            step=0.05,
            value=0.15,
            description="Onboarding Overhead:",
            layout=slider_layout,
        ),
    )
    def update_plot(
        team_size,
        weeks,
        context_switching_factor,
        communication_factor,
        meeting_factor,
        integration_factor,
        onboarding_factor,
    ):
        """Update the plots based on slider values"""

        # Calculate results for current parameters
        result = calculate_unproductive_hours(
            team_size,
            weeks,
            context_switching_factor,
            communication_factor,
            meeting_factor,
            integration_factor,
            onboarding_factor,
        )

        # Calculate results for varying team sizes
        team_sizes = range(1, team_size * 2 + 1)
        team_results = []

        for ts in team_sizes:
            team_result = calculate_unproductive_hours(
                ts,
                weeks,
                context_switching_factor,
                communication_factor,
                meeting_factor,
                integration_factor,
                onboarding_factor,
            )
            team_results.append(team_result)

        # Extract data for plotting
        productivity_percentages = [r["productivity_percentage"] for r in team_results]

        # Create breakdown plot for current parameters
        overhead_categories = [
            "Context Switching",
            "Communication",
            "Meetings",
            "Integration",
            "Onboarding",
            "Productive Hours",
        ]

        hours_data = [
            result["context_switching_hours"],
            result["communication_hours"],
            result["meeting_hours"],
            result["integration_hours"],
            result["onboarding_hours"],
            result["productive_hours"],
        ]

        colors = ["#ff9999", "#ffcc99", "#ffff99", "#99ff99", "#99ccff", "#66ff66"]

        # Create pie chart for time breakdown
        fig1 = go.Figure(
            data=[
                go.Pie(
                    labels=overhead_categories,
                    values=hours_data,
                    marker=dict(colors=colors),
                    textinfo="label+percent",
                    hole=0.3,
                )
            ]
        )

        fig1.update_layout(
            title=f"Time Breakdown (Team Size: {team_size}, Weeks: {weeks})", height=500
        )

        # Create line chart for productivity vs team size
        fig2 = go.Figure()

        fig2.add_trace(
            go.Scatter(
                x=list(team_sizes),
                y=productivity_percentages,
                mode="lines+markers",
                name="Productivity %",
                line=dict(color="green", width=3),
                marker=dict(size=8),
            )
        )

        # Add a vertical line at current team size
        fig2.add_shape(
            type="line",
            x0=team_size,
            y0=0,
            x1=team_size,
            y1=100,
            line=dict(color="red", width=2, dash="dash"),
        )

        fig2.update_layout(
            title="Productivity Percentage vs Team Size",
            xaxis_title="Team Size",
            yaxis_title="Productivity Percentage",
            yaxis=dict(range=[0, 100]),
            height=500,
        )

        # Display figures
        display(fig1)
        display(fig2)

        # Print summary statistics
        print(
            f"\n--- Summary for Team Size: {team_size}, Project Duration: {weeks} weeks ---"
        )
        print(f"Total Available Hours: {result['total_hours']:.0f}")
        print(
            f"Productive Hours: {result['productive_hours']:.0f} ({result['productivity_percentage']:.1f}%)"
        )
        print(
            f"Total Unproductive Hours: {result['total_unproductive']:.0f} ({100-result['productivity_percentage']:.1f}%)"
        )
        print("\nBreakdown of Unproductive Hours:")
        print(
            f"  - Context Switching: {result['context_switching_hours']:.0f} hours ({result['context_switching_hours']/result['total_hours']*100:.1f}%)"
        )
        print(
            f"  - Communication: {result['communication_hours']:.0f} hours ({result['communication_hours']/result['total_hours']*100:.1f}%)"
        )
        print(
            f"  - Meetings: {result['meeting_hours']:.0f} hours ({result['meeting_hours']/result['total_hours']*100:.1f}%)"
        )
        print(
            f"  - Integration: {result['integration_hours']:.0f} hours ({result['integration_hours']/result['total_hours']*100:.1f}%)"
        )
        print(
            f"  - Onboarding: {result['onboarding_hours']:.0f} hours ({result['onboarding_hours']/result['total_hours']*100:.1f}%)"
        )

        # Calculate optimal team size based on maximum productivity
        optimal_idx = np.argmax(productivity_percentages)
        optimal_team_size = team_sizes[optimal_idx]
        optimal_productivity = productivity_percentages[optimal_idx]

        print(
            f"\nOptimal Team Size for this project: {optimal_team_size} (Productivity: {optimal_productivity:.1f}%)"
        )


# Function to analyze the productivity model for different team sizes and project durations
def analyze_productivity_trends():
    """Analyze how productivity changes with team size and project duration"""

    # Define ranges for analysis
    team_sizes = np.arange(1, 31)
    weeks_values = [4, 12, 26, 52]

    # Use default overhead factors
    context_switching_factor = 0.05
    communication_factor = 0.03
    meeting_factor = 0.1
    integration_factor = 0.02
    onboarding_factor = 0.15

    # Calculate productivity for all combinations
    results = []

    for weeks in weeks_values:
        productivity_values = []
        for team_size in team_sizes:
            result = calculate_unproductive_hours(
                team_size,
                weeks,
                context_switching_factor,
                communication_factor,
                meeting_factor,
                integration_factor,
                onboarding_factor,
            )
            productivity_values.append(result["productivity_percentage"])
        results.append((weeks, productivity_values))

    # Create plot
    fig = go.Figure()

    for weeks, productivity_values in results:
        fig.add_trace(
            go.Scatter(
                x=team_sizes,
                y=productivity_values,
                mode="lines+markers",
                name=f"{weeks} weeks",
            )
        )

    fig.update_layout(
        title="Productivity vs Team Size for Different Project Durations",
        xaxis_title="Team Size",
        yaxis_title="Productivity Percentage",
        yaxis=dict(range=[0, 100]),
        height=600,
        width=800,
        legend=dict(x=0.01, y=0.99, bgcolor="rgba(255,255,255,0.8)"),
    )

    return fig
